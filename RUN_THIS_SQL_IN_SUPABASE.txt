-- Add resume storage columns to candidate_profiles table
ALTER TABLE candidate_profiles
ADD COLUMN IF NOT EXISTS resume_file TEXT,
ADD COLUMN IF NOT EXISTS resume_filename TEXT,
ADD COLUMN IF NOT EXISTS resume_filetype TEXT;

-- Verify columns were added
SELECT column_name, data_type 
FROM information_schema.columns 
WHERE table_name = 'candidate_profiles' 
  AND column_name IN ('resume_file', 'resume_filename', 'resume_filetype');
